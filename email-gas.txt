function escapeHtml(str) {
  if (!str) return '';
  return str.replace(/&/g,'&amp;').replace(/</g,'&lt;').replace(/>/g,'&gt;').replace(/"/g,'&quot;');
}

function formatItems(items) {
  if (!items) return '<tr><td colspan="2">-</td></tr>';
  return items.split('\n').filter(function(l){return l.trim()}).map(function(l){
    var parts = l.split('=');
    return '<tr><td>' + escapeHtml(parts[0]) + '</td><td>' + escapeHtml(parts[parts.length-1]) + '</td></tr>';
  }).join('');
}
